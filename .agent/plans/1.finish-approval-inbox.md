### 1b. Update `getActions` in the API client

**File:** `ui/src/api.ts`

Replace the current single-arg `getActions(status)` with a filters object:

```ts
export interface ActionFilters {
  status?: string;
  agent_slug?: string;
  action_type?: string;
}

export function getActions(filters: ActionFilters = {}): Promise<Action[]> {
  const params = new URLSearchParams();
  if (filters.status) params.set('status', filters.status);
  if (filters.agent_slug) params.set('agent_slug', filters.agent_slug);
  if (filters.action_type) params.set('action_type', filters.action_type);
  const qs = params.toString();
  return apiFetch<Action[]>(`/actions${qs ? `?${qs}` : ''}`);
}
```
